1. Создали новый проект в PyCharm
2. pip install django - добавление пакета django
3. python.exe -m pip install --upgrade pip - обновили pip
4. pip freeze > requirements.txt - создали файл requirements.txt с установленными пакетами (потом обновим данные)
5. django-admin startproject orders .
6. django-admin startapp backend
7. python manage.py makemigrations - миграции 
8. python manage.py migrate - создаем базу данных
9. python manage.py runserver - запускаем проект

python manage.py createsuperuser - супер пользователь
